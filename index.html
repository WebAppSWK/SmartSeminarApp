// ========================================== // ระบบสารสนเทศสัมมนาทางการบริหาร - FEU 2026 // ระบบหลังบ้าน (Backend: Google Apps Script) // ========================================== const DRIVE_LINK = "https://drive.google.com/drive/folders/1OYes6tXWqZA7YpQr-wjsYmW7xZYjIjWJ?usp=drive_link"; // ========================================== // 1. ฟังก์ชันรับข้อมูลแบบ GET (ดึงข้อมูลไปแสดงผลที่หน้าเว็บ) // ========================================== function doGet(e) { var action = e.parameter.action; // โหมดสำหรับดึงข้อมูลเพื่อโชว์ที่ Dashboard แผงควบคุม (V1.9) if (action === 'getAdminData') { var ss = SpreadsheetApp.getActiveSpreadsheet(); // ดึงข้อมูลผู้ใช้งาน var userSheet = ss.getSheetByName("users"); if (!userSheet) return ContentService.createTextOutput(JSON.stringify({success: false, error: "ไม่พบแผ่นงาน users"})).setMimeType(ContentService.MimeType.JSON); var userData = userSheet.getDataRange().getValues(); // ดึงข้อมูลการประเมิน var evalSheet = ss.getSheetByName("evaluations"); if (!evalSheet) return ContentService.createTextOutput(JSON.stringify({success: false, error: "ไม่พบแผ่นงาน evaluations"})).setMimeType(ContentService.MimeType.JSON); var evalData = evalSheet.getDataRange().getValues(); var evalList = []; var userScores = {}; var userUniqueBooths = {}; // อ่านข้อมูลการประเมิน (ข้ามหัวตารางแถวที่ 1) for (var j = 1; j < evalData.length; j++) { var studentId = String(evalData[j][1]); // คอลัมน์ B: student_id var studentName = evalData[j][2]; // คอลัมน์ C: student_name var boothId = evalData[j][3]; // คอลัมน์ D: booth_id var rating = evalData[j][4]; // คอลัมน์ E: rating var comment = evalData[j][5]; // คอลัมน์ F: comment evalList.push({ studentId: studentId, studentName: studentName, boothId: boothId, rating: rating, comment: comment }); // นับว่านักศึกษาแต่ละคนเข้าประเมินกี่บูธ (นับเฉพาะบูธที่ไม่ซ้ำกัน) if (!userUniqueBooths[studentId]) { userUniqueBooths[studentId] = []; } if (userUniqueBooths[studentId].indexOf(boothId) === -1) { userUniqueBooths[studentId].push(boothId); } } // สรุปข้อมูลยอดรวมของนักศึกษา var userList = []; for (var i = 1; i < userData.length; i++) { var id = String(userData[i][0]); var name = userData[i][1]; userList.push({ studentId: id, name: name, score: userUniqueBooths[id] ? userUniqueBooths[id].length : 0 }); } var result = { success: true, totalUsers: userList.length, totalEvals: evalList.length, users: userList, evaluations: evalList }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // โหมดสำหรับดึงชื่อบูธแบบเจาะจง (เมื่อหน้าเว็บส่ง ?booth=B01 มา) var boothId = e.parameter.booth || ""; var boothName = getBoothName(boothId); var resultBooth = { boothId: boothId, boothName: boothName }; return ContentService.createTextOutput(JSON.stringify(resultBooth)) .setMimeType(ContentService.MimeType.JSON); } // ========================================== // 2. ฟังก์ชันรับข้อมูลแบบ POST (รับข้อมูลจากหน้าเว็บมาบันทึกในชีท) // ========================================== function doPost(e) { try { var jsonString = e.postData.contents; var data = JSON.parse(jsonString); var studentId = data.studentId; var studentName = data.studentName; var boothId = data.boothId; var rating = data.rating; var comment = data.comment; var result = submitEvaluation(studentId, studentName, boothId, rating, comment); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ========================================== // 3. ฟังก์ชันการทำงานพื้นฐาน (ตัวช่วยย่อย) // ========================================== // ค้นหาชื่อบูธจากแผ่นงาน booths function getBoothName(boothId) { if (!boothId) return ""; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("booths"); if (!sheet) return "ไม่พบข้อมูลบูธ"; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] == boothId) return data[i][1]; } return "ไม่อยู่ในฐานข้อมูล"; } // บันทึกและประมวลผลข้อมูล function submitEvaluation(studentId, studentName, boothId, rating, comment) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- 1. บันทึกข้อมูลนักศึกษา (ถ้ายังไม่มีในระบบ) --- var userSheet = ss.getSheetByName("users"); if (!userSheet) throw new Error("ไม่พบแผ่นงาน users"); var userData = userSheet.getDataRange().getValues(); var userExists = false; for (var i = 1; i < userData.length; i++) { if (String(userData[i][0]) === String(studentId)) { userExists = true; break; } } if (!userExists) { userSheet.appendRow([studentId, studentName]); } // --- 2. บันทึกข้อมูลการประเมิน (อัปเดตถ้าเคยประเมินบูธนี้แล้ว หรือเพิ่มแถวใหม่) --- var evalSheet = ss.getSheetByName("evaluations"); if (!evalSheet) throw new Error("ไม่พบแผ่นงาน evaluations"); var evalData = evalSheet.getDataRange().getValues(); var rowIndexToUpdate = -1; // ค้นหาว่านักศึกษาคนนี้เคยประเมินบูธนี้ไปแล้วหรือยัง? (ถ้าใช่ จะจำเลขแถวไว้เพื่อทับข้อมูลเก่า) for (var j = 1; j < evalData.length; j++) { if (String(evalData[j][1]) === String(studentId) && String(evalData[j][3]) === String(boothId)) { rowIndexToUpdate = j + 1; // +1 เพราะแถวในสเปรดชีตเริ่มที่ 1, และ index เริ่มที่ 0 break; } } var timeStamp = new Date(); if (rowIndexToUpdate !== -1) { // กรณีที่เคยประเมินแล้ว: ทำการอัปเดตข้อมูลทับแถวเดิม (เพื่อป้องกันข้อมูลซ้ำซ้อนกันในชีท) evalSheet.getRange(rowIndexToUpdate, 1).setValue(timeStamp); evalSheet.getRange(rowIndexToUpdate, 5).setValue(rating); evalSheet.getRange(rowIndexToUpdate, 6).setValue(comment); } else if (boothId) { // กรณีใหม่: บันทึกข้อมูลแถวใหม่ต่อท้ายเลย // ลำดับคอลัมน์ A:เวลา | B:รหัส | C:ชื่อ | D:บูธ | E:ดาว | F:คอมเมนต์ evalSheet.appendRow([timeStamp, studentId, studentName, boothId, rating, comment]); } return { success: true, message: "บันทึกข้อมูลเรียบร้อย" }; }
